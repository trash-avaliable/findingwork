波次管理器的任务：根据游戏时间，从当前波次配置中**不断生成怪物**，并控制生成位置、数量、精英变异等。
node：WaveManager
### 波次管理器核心逻辑
# wave_manager.gd
extends Node

@onready var player = get_tree().get_first_node_in_group("player")
var current_wave_index: int = 0
var wave_start_time: float = 0.0
var spawn_timer: float = 0.0
var enemies_on_screen: Array[Node] = []   # 用于数量控制

func _process(delta):
    if not player: return
    var current_time = Time.get_ticks_msec() / 1000.0  # 秒为单位
    
    # 1. 更新当前波次索引
    update_current_wave(current_time)
    
    # 2. 生成定时器
    var wave = get_current_wave_config()
    if wave and enemies_on_screen.size() < wave_config.max_enemies_on_screen:
        spawn_timer += delta
        var interval = wave_config.base_spawn_interval
        if spawn_timer >= interval:
            spawn_timer = 0
            try_spawn_enemies(wave)
    
    # 3. 清理已死亡的敌人（从列表中移除）
    enemies_on_screen = enemies_on_screen.filter(func(e): return is_instance_valid(e) and not e.is_queued_for_deletion())

func update_current_wave(current_time):
    for i in range(wave_config.waves.size()):
        var w = wave_config.waves[i]
        if current_time >= w.start_time and current_time < w.end_time:
            current_wave_index = i
            break

func get_current_wave_config():
    if current_wave_index < wave_config.waves.size():
        return wave_config.waves[current_wave_index]
    return null

func try_spawn_enemies(wave_config):
    # 计算本次要生成的总怪物数量（可根据玩家强度动态调整）
    var spawn_count = randi_range(1, 3)   # 基础每次生成1~3只，后面会乘以系数
    for i in range(spawn_count):
        var spawn_config = pick_spawn_config_by_weight(wave_config.spawn_configs)
        if spawn_config:
            var count = randi_range(spawn_config.min_spawn, spawn_config.max_spawn)
            for j in range(count):
                spawn_single_enemy(spawn_config)

func pick_spawn_config_by_weight(spawn_configs: Array) -> Dictionary:
    # 权重随机选择
    var total_weight = 0.0
    for cfg in spawn_configs:
        total_weight += cfg.weight
    var r = randf() * total_weight
    var accum = 0.0
    for cfg in spawn_configs:
        accum += cfg.weight
        if r <= accum:
            return cfg
    return spawn_configs[0]
### 生成单个怪物（包括精英怪和位置策略）
func spawn_single_enemy(spawn_config: Dictionary):
    var enemy_id = spawn_config.enemy_id
    var enemy_data = EnemyDatabase.enemies[enemy_id]
    
    # 精英怪判定
    var is_elite = randf() < spawn_config.get("elite_chance", 0.0)
    
    # 实例化场景
    var scene = load(enemy_data.scene_path)
    var enemy = scene.instantiate()
    enemy.set_enemy_data(enemy_data, is_elite)   # 自定义初始化方法
    
    # 设置生成位置：在玩家视野外（屏幕边缘外扩一定距离）
    var spawn_pos = get_spawn_position_outside_camera()
    enemy.global_position = spawn_pos
    
    # 添加到场景
    get_tree().current_scene.add_child(enemy)
    enemies_on_screen.append(enemy)

func get_spawn_position_outside_camera() -> Vector2:
    var camera = get_viewport().get_camera_2d()
    var cam_center = camera.global_position
    var view_size = get_viewport().get_visible_rect().size
    # 在屏幕外四周随机选一个方向
    var side = randi_range(0, 3)   # 0左 1右 2上 3下
    var offset = 100.0   # 外扩距离
    match side:
        0: return cam_center + Vector2(-view_size.x/2 - offset, randf_range(-view_size.y/2, view_size.y/2))
        1: return cam_center + Vector2(view_size.x/2 + offset, randf_range(-view_size.y/2, view_size.y/2))
        2: return cam_center + Vector2(randf_range(-view_size.x/2, view_size.x/2), -view_size.y/2 - offset)
        3: return cam_center + Vector2(randf_range(-view_size.x/2, view_size.x/2), view_size.y/2 + offset)
    return cam_center

### 对象池优化

为了避免频繁实例化/销毁，可以为普通怪物使用对象池。简单实现：

gdscript

# enemy_pool.gd
var pools = {}   # key = scene_path, value = Array[Node]
func get_enemy(scene_path):
    if not pools.has(scene_path):
        pools[scene_path] = []
    var pool = pools[scene_path]
    if pool.size() > 0:
        return pool.pop_back()
    else:
        return load(scene_path).instantiate()
func return_enemy(enemy):
    enemy.get_parent().remove_child(enemy)
    pools[enemy.scene_file_path].append(enemy)

### 威胁等级计算

每个怪物的 `threat_level` 不是拍脑袋填的，而是基于公式自动生成，然后手动微调。参考公式：

text

threat_level = (hp * damage * speed_factor) / player_dps_baseline

其中 `speed_factor` 根据移速映射（移速越快威胁越大），`player_dps_baseline` 是玩家在该关卡的预期输出（由设计师设定）。
### 动态难度调整（DDA）

游戏应该根据玩家的表现动态调整生成难度，避免过于简单或过于困难。

#### 4.2.1 基于玩家等级的生成权重修正

在 `pick_spawn_config_by_weight` 中增加难度系数：

gdscript

func get_difficulty_factor() -> float:
    var player_level = PlayerStats.level
    var time = Time.get_ticks_msec() / 1000.0
    # 基础系数 1.0，随时间增加，随等级增加
    return 1.0 + time / 300.0 + player_level / 50.0
func pick_spawn_config_by_weight(spawn_configs):
    var total_weight = 0.0
    var difficulty_factor = get_difficulty_factor()
    for cfg in spawn_configs:
        var enemy_data = EnemyDatabase.enemies[cfg.enemy_id]
        # 威胁越高的怪物，在高难度下权重提升越多
        var weight_mod = 1.0 + enemy_data.threat_level * (difficulty_factor - 1.0)
        cfg.current_weight = cfg.weight * weight_mod
        total_weight += cfg.current_weight

#### 4.2.2 动态调整生成数量上限

根据玩家当前屏幕上的怪物数量动态调整生成速率：

gdscript

func get_dynamic_spawn_interval() -> float:
    var base = wave_config.base_spawn_interval
    var ratio = float(enemies_on_screen.size()) / wave_config.max_enemies_on_screen
    # 怪物越多，生成越慢（防止溢出）
    return base + ratio * 0.5

### 4.3 操作系数（玩家技巧补偿）

引入“操作系数”可以奖励高技巧玩家，同时保护低技巧玩家。

#### 4.3.1 无伤奖励

连续无伤时间越长，获得更多经验或金币，但同时怪物略微增强（可选）。实现：

gdscript

# 在玩家受伤时重置
var no_hit_timer: float = 0.0
func _process(delta):
    if not damaged_recently:
        no_hit_timer += delta
        var bonus = clamp(no_hit_timer / 60.0, 0.0, 0.5)   # 最多50%加成
        PlayerStats.exp_multiplier = 1.0 + bonus

#### 4.3.2 连杀奖励

在短时间内连续击杀多个怪物，给予临时增益（如攻击力提升）。

gdscript

var kill_streak: int = 0
var streak_timer: float = 0.0
func on_enemy_killed():
    streak_timer = 3.0   # 重置连杀计时
    kill_streak += 1
    if kill_streak >= 10:
        apply_streak_bonus()
func _process(delta):
    streak_timer -= delta
    if streak_timer <= 0:
        kill_streak = 0
